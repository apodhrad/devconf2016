\subsection{Example}
\begin{frame}[fragile]
\frametitle{How would you debug this?}
\begin{lstlisting}
for (c = 0; c < ( n - 1 ); c++) {
  for (d = 0; d < n - c - 1; d++) {
    if (array[d] > array[d+1]) {
      swap       = array[d];
      array[d]   = array[d+1];
      array[d+1] = swap;
    }
  }
}
\end{lstlisting}
\end{frame}
